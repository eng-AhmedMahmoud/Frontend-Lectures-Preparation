Fluid Typography Function

// Fluid typography Sass function
@function fluid-type($min-size, $max-size, $min-width: 320px, $max-width: 1200px) {
  $slope: ($max-size - $min-size) / ($max-width - $min-width);
  $intercept: $min-size - $slope * $min-width;
  @return clamp(#{$min-size}, #{$intercept} + #{$slope * 100vw}, #{$max-size});
}

// Usage
.text-lg {
  font-size: fluid-type(1rem, 1.25rem);
}

Advanced Button Mixin

// Advanced button variant generator
@mixin button-variant($variant, $style: 'solid') {
  $color-map: map-get($button-colors, $variant);
  $bg: map-get($color-map, 'bg');
  $color: map-get($color-map, 'color');
  $border: map-get($color-map, 'border');
  
  @if $style == 'solid' {
    background-color: $bg;
    color: $color;
    border-color: $border;
  } @else if $style == 'outline' {
    background-color: transparent;
    color: $bg;
    border: 2px solid $bg;
  } @else if $style == 'ghost' {
    background-color: transparent;
    color: $bg;
    border: none;
  }
  
  // Generate hover states
  &:hover {
    @include button-hover-state($variant, $style);
  }
}

// Generate all button variants
@each $variant, $colors in $button-colors {
  @each $style in ('solid', 'outline', 'ghost', 'soft') {
    .btn--#{$variant}#{if($style != 'solid', '-#{$style}', '')} {
      @include button-variant($variant, $style);
    }
  }
}
